from scapy.all import *

print("[*] SOC Packet Analyzer Started...")

def analyze(packet):
    # Check if packet has IP layer
    if packet.haslayer(IP):
        src = packet[IP].src
        dst = packet[IP].dst

        output = f"{src} → {dst}"
        print(output)

        # Save to log file
        with open("log.txt", "a") as f:
            f.write(output + "\n")

# Start sniffing
sniff(prn=analyze, count=50, store=False)
